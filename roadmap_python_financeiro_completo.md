  ```python
  def sma(data, period=20):
      return data.rolling(window=period).mean()

  def rsi(data, period=14):
      delta = data.diff()
      gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
      loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
      rs = gain / loss
      return 100 - (100 / (1 + rs))
  ```

  ```python
      predictions = []
      actuals = []

      for i in range(window_size, len(data)):
          # Treinar com dados até i-1
          train_data = data.iloc[i-window_size:i-1]
          test_data = data.iloc[i:i+1]

          # Fazer predição
          pred = model.predict(test_data.drop('target', axis=1))
  ```

  ```python
  from numba import jit, njit
  import numpy as np
  import time

  # SMA com Numba (100x mais rápido)
  @njit
  def fast_sma(prices, window):
      """SMA otimizado com Numba JIT"""
      n = len(prices)
      sma = np.zeros(n)

      for i in range(window, n):
          sma[i] = np.mean(prices[i-window:i])

      return sma

  # RSI com Numba
  @njit
  def fast_rsi(prices, window=14):
      """RSI otimizado com Numba"""
      n = len(prices)
      rsi = np.zeros(n)
      deltas = np.diff(prices)

      gains = np.where(deltas > 0, deltas, 0)
      losses = np.where(deltas < 0, -deltas, 0)

      avg_gain = np.mean(gains[:window])
      avg_loss = np.mean(losses[:window])

      for i in range(window, n):
          avg_gain = (avg_gain * (window - 1) + gains[i-1]) / window
          avg_loss = (avg_loss * (window - 1) + losses[i-1]) / window

          rs = avg_gain / avg_loss if avg_loss != 0 else 0
          rsi[i] = 100 - (100 / (1 + rs))

      return rsi

  # Bollinger Bands com Numba
  @njit
  def fast_bollinger_bands(prices, window=20, num_std=2):
      """Bollinger Bands otimizadas com Numba"""
      n = len(prices)
      sma = fast_sma(prices, window)
      std = np.zeros(n)

      for i in range(window, n):
          std[i] = np.std(prices[i-window:i])

      upper = sma + (std * num_std)
      lower = sma - (std * num_std)

      return sma, upper, lower

  # Performance comparison
  def benchmark_indicators(prices):
  ```

  ```python
  @njit
  def monte_carlo_option(S0, K, r, sigma, T, simulations=100000):
      """Black-Scholes Monte Carlo ultra-rápido com Numba"""

      # Gerar números aleatórios
      z = np.random.standard_normal(simulations)

      # Simular preço final
      ST = S0 * np.exp((r - 0.5 * sigma**2) * T + sigma * np.sqrt(T) * z)

      # Calcular payoff para call option
      payoff = np.maximum(ST - K, 0)

      # Valor presente esperado
      option_value = np.exp(-r * T) * np.mean(payoff)

      return option_value

  @njit
  def monte_carlo_asian_option(S0, K, r, sigma, T, time_steps=252, simulations=10000):
      """Asian Option Monte Carlo com Numba"""

      dt = T / time_steps
      option_values = np.zeros(simulations)

      for sim in range(simulations):
          prices = np.zeros(time_steps + 1)
          prices[0] = S0

          # Simular caminho do preço
          for t in range(1, time_steps + 1):
              z = np.random.standard_normal()
              prices[t] = prices[t-1] * np.exp((r - 0.5 * sigma**2) * dt + sigma * np.sqrt(dt) * z)

          # Payoff para Asian option (média dos preços)
          avg_price = np.mean(prices)
          payoff = np.maximum(avg_price - K, 0)
          option_values[sim] = np.exp(-r * T) * payoff

      return np.mean(option_values)

  # Exemplo de uso
  def test_monte_carlo():
      """Testa Monte Carlo com Numba"""

      # Parâmetros
      S0 = 100      # Preço inicial
      K = 105       # Strike
      r = 0.05      # Taxa livre de risco
      sigma = 0.2   # Volatilidade
      T = 1.0       # Tempo até expiração (1 ano)

      # Testar European option
      start = time.time()

      # Testar Asian option
  ```

  ```python
  @njit
  def high_frequency_backtest(prices, volumes, tick_size=0.01, latency_ms=1):
      """Backtest de alta frequência com Numba"""

      n = len(prices)
      positions = np.zeros(n)
      cash = np.zeros(n)
      portfolio_value = np.zeros(n)

      # Parâmetros
      initial_cash = 10000
      cash[0] = initial_cash
      max_position = 100
      lookback = 100

      for i in range(lookback, n):
          # Estratégia: momentum de curto prazo
          recent_prices = prices[i-lookback:i]
          recent_returns = np.diff(recent_prices) / recent_prices[:-1]

          if len(recent_returns) > 0:
              momentum = np.mean(recent_returns[-10:])  # Últimos 10 períodos
              volatility = np.std(recent_returns[-20:])    # Últimos 20 períodos

              # Sinais
              if momentum > 0.001 and volatility < 0.02:  # Momentum positivo + baixa vol
                  signal = 1
              elif momentum < -0.001 and volatility < 0.02:  # Momentum negativo + baixa vol
                  signal = -1
              else:
                  signal = 0

              # Executar trade
              if signal != 0 and positions[i-1] == 0:
                  # Abrir posição
                  trade_size = min(max_position, int(cash[i-1] / prices[i]))
                  positions[i] = signal * trade_size
                  cash[i] = cash[i-1] - signal * trade_size * prices[i]

              elif signal == 0 and positions[i-1] != 0:
                  # Fechar posição
                  cash[i] = cash[i-1] + positions[i-1] * prices[i]
                  positions[i] = 0
              else:
                  # Manter posição
                  positions[i] = positions[i-1]
                  cash[i] = cash[i-1]
          else:
              positions[i] = positions[i-1]
              cash[i] = cash[i-1]

          # Valor do portfólio
          portfolio_value[i] = cash[i] + positions[i] * prices[i]

      # Métricas
      returns = np.diff(portfolio_value) / portfolio_value[:-1]
      total_return = (portfolio_value[-1] - initial_cash) / initial_cash
      sharpe_ratio = np.sqrt(252) * np.mean(returns) / np.std(returns) if np.std(returns) > 0 else 0
      max_drawdown = np.max(portfolio_value) - np.min(portfolio_value)

      return {
          "total_return": total_return,
          "sharpe_ratio": sharpe_ratio,
          "max_drawdown": max_drawdown,
          "final_value": portfolio_value[-1],
          "portfolio_values": portfolio_value
      }
  ```

  ```python
  @njit
  def efficient_frontier(returns, num_portfolios=1000):
      """Fronteira Eficiente otimizada com Numba"""

      n_assets = returns.shape[1]
      mean_returns = np.mean(returns, axis=0)
      cov_matrix = np.cov(returns.T)

      # Arrays para resultados
      portfolio_returns = np.zeros(num_portfolios)
      portfolio_volatilities = np.zeros(num_portfolios)
      portfolio_weights = np.zeros((num_portfolios, n_assets))

      for i in range(num_portfolios):
          # Gerar pesos aleatórios
          weights = np.random.random(n_assets)
          weights = weights / np.sum(weights)

          # Calcular retorno e volatilidade
          portfolio_return = np.sum(mean_returns * weights) * 252
          portfolio_volatility = np.sqrt(np.dot(weights.T, np.dot(cov_matrix, weights))) * np.sqrt(252)

          portfolio_returns[i] = portfolio_return
          portfolio_volatilities[i] = portfolio_volatility
          portfolio_weights[i] = weights

      return portfolio_returns, portfolio_volatilities, portfolio_weights

  @njit
  def max_sharpe_portfolio(returns, risk_free_rate=0.02):
      """Portfólio de máximo Sharpe com Numba"""

      port_returns, port_vols, port_weights = efficient_frontier(returns, 10000)

      # Calcular Sharpe ratios
      sharpe_ratios = (port_returns - risk_free_rate) / port_vols

      # Encontrar máximo Sharpe
      max_idx = np.argmax(sharpe_ratios)

      return {
          "weights": port_weights[max_idx],
          "return": port_returns[max_idx],
        "volatility": port_vols[max_idx],
        "sharpe_ratio": sharpe_ratios[max_idx]
      }

  # Exemplo de uso
  def test_portfolio_optimization():
  ```

  ```python
  class ProductionStrategy:

              data = self.get_market_data()

              signals = self.generate_signals(data, strategy_config)

      def log_trade(self, trade):
  ```

  ```python
              'strategies': self.check_strategies
          }

          status = {}
          overall_healthy = True

          for name, check_func in self.checks.items():
              try:
                  result = check_func()
                  status[name] = {'status': 'healthy', 'details': result}
              except Exception as e:
                  status[name] = {'status': 'unhealthy', 'error': str(e)}
                  overall_healthy = False

          return {
              'status': 'healthy' if overall_healthy else 'unhealthy',
              'timestamp': datetime.utcnow().isoformat(),
              'checks': status
  ```
